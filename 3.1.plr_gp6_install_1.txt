######## Greenplum plr installation
## 3.1.1 plr installation
$ gppkg --install plr-3.1.3-gp6-rhel9-x86_64.gppkg
$ gpstop -af
$ source $GPHOME/greenplum_path.sh
$ gpstart -a
$ psql -c "CREATE EXTENSION plr;"

## 3.1.2 
$ gppkg --install DataScienceR-2.2.0-gp6-rhel9_x86_64.gppkg
$ source $GPHOME/greenplum_path.sh
$ gpstop -af
$ gpstart -a

## 3.1.3.1
## Change permissions to superuser so that only the udba user can execute plr.
$ psql -c "ALTER ROLE udba SUPERUSER; "
