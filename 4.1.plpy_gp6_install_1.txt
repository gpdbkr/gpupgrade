######## Greenplum plpython installation
## 4.1.1 plpython3 installation
$ psql -c "CREATE EXTENSION plpython3u;"

## 4.1.2 
$ gppkg --install DataSciencePython3.9-2.2.6-gp6-rhel9_x86_64.gppkg
$ gpconfig -c plpython3.python_path \
  -v "'$GPHOME/ext/DataSciencePython3.9/lib/python3.9/site-packages:$GPHOME/ext/DataSciencePython3.9/lib64/python3.9/site-packages'" \
  --skipvalidation
$ gpstop -u

## 4.1.3.1
## Change permissions to superuser so that only the udba user can execute plr.
$ psql -c "ALTER ROLE udba SUPERUSER; "
